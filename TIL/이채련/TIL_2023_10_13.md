## 기본 input 태그를 활용하면 자동으로 제공됨!!

- type=”file”을 활용한다

```jsx
<input
        name="addPicture"
        **type="file"**
        accept="image/*;capture=camera"
        className="w-[147px] h-[147px] opacity-0"
        ref={imgRef}
      />
```

## 업로드하자마자 미리보여주기

1. input type=”file”로 입력받는다
2. FileReader로 읽는다
3. img태그로 띄워준다

```jsx
// input으로 입력받을 ref
const **imgRef** = useRef<HTMLInputElement | null>(null);
// 띄워줄 이미지를 저장할 상태
const [img, setImg] = useState<string | ArrayBuffer | null>(null);

// input type="file"로 입력받은 값을 파일이다. 따라서 바로 렌더링 해주려면
// FileReader를 이용해서 읽어줘야함
const saveImgFile = () => {
    if (imgRef.current !== null && imgRef?.current?.files !== null) {
      const file = imgRef.current.files[0];
      const reader = new FileReader();
      reader.readAsDataURL(file);
      reader.onloadend = () => {
        // 인풋으로 입력받은 파일을 읽어서 img 상태에 저장!
        setImg(reader.result);
      };
    }
  };

// input으로 이미지 파일을 입력받고
<input
      name="addPicture"
      type="file"
      accept="image/*;capture=camera"
      className="w-[147px] h-[147px] opacity-0"
      onChange={saveImgFile}
      ref={**imgRef**}
    />

// 이미지 태그의 src에 img를 넣어주면 렌더링된다!
<img
      // 렌더링 할 수 있게 문자열로 변환시켜서 렌더링하기
      src={img.toString()}
      alt="Upload Img"
      className="w-[100%] h-[100%] object-cover rounded-lg"
    />
```
